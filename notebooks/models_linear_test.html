# validate model with tgs
for istat,stat in enumerate(load_cfsr_moana_uhslc.predictand_reduced.tg_names.values):
    ss_moana = load_cfsr_moana_uhslc.predictand_reduced.isel(site=istat)\
        .load().resample(time='1D').max()
    stats_stat = MultiLinear_Regression(
        PCA_DynamicPred(
            load_cfsr_moana_uhslc.predictor_slp,
            winds=(True,load_cfsr_moana_uhslc.predictor_wind),
            wind_vars=('wind_proj_mask','lon','lat'),
            region=(True,(
                ss_moana.lon.values[0]-1.5,
                ss_moana.lon.values[0]+1.5,
                ss_moana.lat.values[0]+1.5,
                ss_moana.lat.values[0]-1.5
            )),
            pca_plot=(True,True),
            pca_ttls=['SLP in t', 'WINDs in t']
        )[0], ss_moana, plot_results=True,
        validator=(
            True,load_cfsr_moana_uhslc.validator.isel(name=istat),'ss'
        )
    )
